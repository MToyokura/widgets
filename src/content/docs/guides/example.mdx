## JSXGraph

<JSXGraph id="myBoard" height="450px" />

<script
  is:inline
  set:html={`
  window.initJSXGraph = (id) => {
    // 1. Initialize the board
    const board = JXG.JSXGraph.initBoard(id, {
      boundingbox: [-5, 5, 5, -5],
      axis: true,
      showCopyright: true,
    });

    // 2. Create Geometry
    const p1 = board.create('point', [-2, 1], { name: 'A', color: 'blue' });
    const p2 = board.create('point', [2, -1], { name: 'B', color: 'red' });
    
    // 3. Create a line through the points
    board.create('line', [p1, p2], { strokeColor: 'green', dash: 2 });
    
    // 4. Create a circle based on the distance between points
    board.create('circle', [p1, p2], { strokeColor: '#00ff00', fillOpacity: 0.1, fillColor: 'green' });
  };
`}
></script>
